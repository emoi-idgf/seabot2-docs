155static inline float FusionAsin(const float value)
156{
157 if (value <= -1.0f) {
158 return (float) M_PI / -2.0f;
159 }
160 if (value >= 1.0f) {
161 return (float) M_PI / 2.0f;

177static inline float FusionFastInverseSqrt(const float x)
178{
179
180 typedef union {
181 float f;
182 int32_t i;
183 } Union32;
184
185 Union32 union32 = {.f = x};
186 union32.i = 0x5F1F1412 - (union32.i >> 1);
187 return union32.f * (1.69000231f - 0.714158168f * x * union32.f * union32.f);
188}

477static inline FusionMatrix FusionQuaternionToMatrix(const FusionQuaternion quaternion)
478{
479#define Q quaternion.element
480 const float qwqw = Q.w * Q.w; // calculate common terms to avoid repeated operations
481 const float qwqx = Q.w * Q.x;
482 const float qwqy = Q.w * Q.y;
483 const float qwqz = Q.w * Q.z;
484 const float qxqy = Q.x * Q.y;
485 const float qxqz = Q.x * Q.z;
486 const float qyqz = Q.y * Q.z;
487 const FusionMatrix matrix = {.element = {
488 .xx = 2.0f * (qwqw - 0.5f + Q.x * Q.x),
489 .xy = 2.0f * (qxqy - qwqz),
490 .xz = 2.0f * (qxqz + qwqy),
491 .yx = 2.0f * (qxqy + qwqz),
492 .yy = 2.0f * (qwqw - 0.5f + Q.y * Q.y),
493 .yz = 2.0f * (qyqz - qwqx),
494 .zx = 2.0f * (qxqz - qwqy),
495 .zy = 2.0f * (qyqz + qwqx),
496 .zz = 2.0f * (qwqw - 0.5f + Q.z * Q.z),
497 }};
498 return matrix;
499#undef Q
500}

507static inline FusionEuler FusionQuaternionToEuler(const FusionQuaternion quaternion)
508{
509#define Q quaternion.element
510 const float halfMinusQySquared = 0.5f - Q.y * Q.y; // calculate common terms to avoid repeated operations
511 const FusionEuler euler = {.angle = {
512 .roll = FusionRadiansToDegrees(atan2f(Q.w * Q.x + Q.y * Q.z, halfMinusQySquared - Q.x * Q.x)),
513 .pitch = FusionRadiansToDegrees(FusionAsin(2.0f * (Q.w * Q.y - Q.z * Q.x))),
514 .yaw = FusionRadiansToDegrees(atan2f(Q.w * Q.z + Q.x * Q.y, halfMinusQySquared - Q.z * Q.z)),
515 }};
516 return euler;
517#undef Q
518}
