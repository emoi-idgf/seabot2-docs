264unsigned int LogData::serialize_data(T &bits, const unsigned int &nb_bit, const unsigned int &start_bit, const double &value, const double &value_min, const double &value_max){
265 double scale = ((1<<nb_bit)-1)/(value_max-value_min);
266 double bit_max = (1<<nb_bit)-1;
267 long unsigned int v = static_cast<long unsigned int>(std::min(std::max(round((value-value_min)*scale), 0.0), bit_max));
268 auto v_max = static_cast<long unsigned int>(bit_max);
269 v = std::min(v, v_max);
270
271 T mask = ((T(1)<<nb_bit)-1);
272 bits &= ~(mask << start_bit);
273 bits |= (T(v) & mask) << start_bit;
274 return nb_bit;
275}
