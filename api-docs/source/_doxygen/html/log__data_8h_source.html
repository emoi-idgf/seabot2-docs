297template < typename T >
298unsigned int LogData::serialize_data(
299 T & bits, const unsigned int & nb_bit,
300 const unsigned int & start_bit, const double & value, const double & value_min,
301 const double & value_max)
302{
303 double scale = ((1 << nb_bit) - 1) / (value_max - value_min);
304 double bit_max = (1 << nb_bit) - 1;
305 long unsigned int v = static_cast < long unsigned int >
306 (std::min(std::max(round((value - value_min) * scale), 0.0), bit_max));
307 auto v_max = static_cast < long unsigned int > (bit_max);
308 v = std::min(v, v_max);
309
310 T mask = ((T(1) << nb_bit) - 1);
311 bits &= ~(mask << start_bit);
312 bits |= (T(v) & mask) << start_bit;
313 return nb_bit;
314}
